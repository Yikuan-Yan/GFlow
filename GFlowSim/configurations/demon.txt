// This is a 2d simulation
Dimensions: 2

Var: phi=0.025
Var: width=20   // Box width
Var: length=0.5 // Door area (length)
Var: tau=0.25
Var: Tl=0.001
Var: Tr=0.001
Var: rhoL=3
Var: rhoR=3
Var: type=0
Var: force=HardSphere
Var: radius=0.05 // Radius of the gas particles.
Var: sigma=0.05 // Radius of the wall particles.

// Set the simulation bounds - a box with a given width.
Bounds: {
  : 0, 2*width
  : 0, width
}

Integrator: VelocityVerlet

Boundary: Repulse

// Set the number of particle types
NTypes: 4

// Creates a random force network of HardSphere and Lennard-Jones forces
Force-grid: {
  : 0, 0, force
  : 0, 1, force
  : 1, 1, force

  : 0, 2, HardSphere
  : 1, 2, HardSphere
  : 0, 3, DemonWall
  : 1, 3, DemonWall
}

Template: TypeA {
  Radius: radius
  Mass: Density=1
  Type: 0
}

Template: TypeB {
  Radius: radius
  Mass: Density=1
  Type: 1
}

// Fill the left side
Fill: Area {
  // Fill all the simulation with particles
  Bounds: {
    : 0, width - 0.05
    : 0, width
  }

  Number: Rho=rhoL {
    TypeA : 1
  }

  Velocity: Temperature=Tl
}

// Fill the right side
Fill: Area {
  // Fill all the simulation with particles
  Bounds: {
    : width + 0.05, 2*width
    : 0, width
  }

  Number: Rho=rhoR {
    TypeB : 1
  }

  Velocity: Temperature=Tr
}

// Top part of the partition
Fill: Wall {
  Start: width, width
  End: width, width/2 + length/2 + sigma
  Radius: sigma
  Type: 2
}

// Bottom part of the partition
Fill: Wall {
  Start: width, width/2 - length/2 - sigma
  End: width, 0
  Radius: sigma
  Type: 2
}

// The demon door
Fill: Wall {
  Start: width, width/2 + length/2 + radius
  End: width, width/2 - length/2 - sigma
  Radius: sigma
  Type: 3

  Modifier: Demon=1 {
    Tau: tau
    Type: type
  }
}
