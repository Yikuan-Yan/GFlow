// Variables
Var: dimensions=2
Var: width=16
Var: length=10
Var: number=1
Var: phi=0.5
Var: corr=0
Var: phiF=0.1

Dimensions: dimensions

// Set the simulation bounds
Bounds: Box=width

Integrator: NoseHooverVelocityVerlet

Boundary: Wrap

// Set the number of particle types
// 0, 3 : Polymers A, B - primary particles.
// 1 : Chain
// 2 : Solvent
NTypes: 4

Force-grid: {
  : 0, 0, HardSphere // Polymer - Polymer  
  : 0, 1, None // Polymer - Chain
  : 0, 2, HardSphere {
    Repulsion: 500
  }
  : 0, 3, HardSphere

  : 1, 1, HardSphere // Chain - Chain

  : 2, 2, HardSphere // Solvent - Solvent
  
  : 3, 1, None // Polymer - Chain
  : 3, 2, HardSphere
  : 3, 3, None // Polymer - Polymer
}

Fill: Polymer {
  Length: length
  Number: number
  R: 0.05
  r: 0.02
  Phi: phi
  Correlation: corr
}

// Fill an area with particles
Fill: Area {
  // Fill all the simulation with particles
  Bounds: Full
  
  Template: TypeA {
    Radius: 0.05
    Mass: Density=1
    Type: 2
  }

  Number: Phi=phiF {
    TypeA : 1
  }

  Velocity: Temperature=0.001
}
