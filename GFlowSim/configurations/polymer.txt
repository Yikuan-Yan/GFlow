Var: number=1

// Set the number of particle types
// 0, 3 : Polymers A, B - primary particles.
// 1 : Chain
// 2 : Solvent

Force-grid: {
  : 0, 0, None  

  : 0, 2, HardSphere
  : 0, 3, HardSphere
  : 1, 1, HardSphere
  : 2, 2, HardSphere
  : 2, 3, HardSphere

  : 3, 3, None
  : 3, 3, HardSphere
}

Fill: Polymer {
  Length: length
  Number: number
  R: 0.05
  r: 0.018 // 0.01
  Phi: phi
  Correlation: corr
  Special: 1
}

  // Fill all the simulation with particles
  Bounds: Full
  
  Template: TypeA {
    Radius: 0.05
    Mass: Density=1
    Type: 2
  }

  Number: Phi=0.1 {
    TypeA : 1
  }
