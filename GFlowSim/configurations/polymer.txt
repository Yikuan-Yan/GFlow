// Variables
Var: dimensions=2
Var: width=16
Var: length=10
Var: number=1
Var: phi=0.5
Var: corr=0

Dimensions: dimensions

// Set the simulation bounds
Bounds: Box=width

Integrator: VelocityVerlet

Boundary: Wrap

// Set the number of particle types
NTypes: 3

// Creates a random force network of HardSphere and Lennard-Jones forces
Force-grid: {
  : 0, 0, HardSphere
  : 0, 2, HardSphere
  : 2, 2, HardSphere
  : 1, 1, HardSphere
  : 1, 0, None
  : 1, 2, None
}

Fill: Polymer {
  Length: length
  Number: number
  R: 0.05
  r: 0.022 // 0.01
  Phi: phi
  Correlation: corr
}

// Fill an area with particles
Fill: Area {
  // Fill all the simulation with particles
  Bounds: Full
  
  Template: TypeA {
    Radius: 0.05
    Mass: Density=1
    Type: 2
  }

  Number: Phi=0.1 {
    TypeA : 1
  }

  Velocity: Temperature=0.001
}
