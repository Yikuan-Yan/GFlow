// Granular "wind tunnel" simulation
// Min radius tries was 0.0125-0.015. Needed a max dt of 0.0002.

// This is a 3d simulation
Dimensions: 3

// VARIABLES
Var: MinR=0.1
Var: MaxR=0.12
Var: Velocity=2
Var: Length=6
Var: Front=4
Var: Width=4

// Set the simulation bounds
Bounds: {
  : -Front, Length
  : -Width, Width
  : -Width, Width
}

Integrator: VelocityVerlet

Boundary: {
  : Wrap
  : Repulse
  : Repulse
}

// Set the number of particle types
NTypes: 1

// Creates a random force network of HardSphere and Lennard-Jones forces
Force-grid: Hertz

// Fill an area with particles
Fill: Area {
  // Fill all the simulation with particles
  Bounds: Full

  Template: TypeA {
    Radius: Uniform {
      Min: MinR
      Max: MaxR
    }
    Mass: Density=1
    Type: 0
  }

  Number: Phi=0.74 { // 3D max packing ~0.74
    TypeA : 1
  }

  Velocity: Zero
}

Particle {
  Position: 0, 0, 0
  Sigma: 1
  Velocity: 0, 0, 0
  Mass: inf
}

Reconcile: Remove=0.5

Modifier: WindTunnel=Velocity
