// This is a 2d simulation
Dimensions: 2

// Set the simulation bounds
Bounds: {
  : -25, 25 // X
  : -25, 25 // Y
}

Integrator: LangevinIntegrator {
  Temperature: 0
}

Boundary: Wrap

Attraction: 1

// Set the number of particle types
NTypes: 1

// Creates a random force network of HardSphere and Lennard-Jones forces
Force-grid: {
  : 0, 0, HardSphereGeneral
}

// Fill an area with particles
Fill: Area {
  // Fill all the simulation with particles
  Bounds: Sphere {
    : 0, 0 // Centered
    : 15
  }

  Template: TypeA {
    Radius: Uniform {
      Min: 0.04
      Max: 0.06
    }

    Mass: Density=1

    Type: 0
  }

  Number: Phi=0.95 {
    TypeA : 1
  }

  Velocity: Zero

  Attraction: 1
}

// The large, dropping particle
Particle {
  Position: 0, 25
  Sigma: 2
  Velocity: 0, -10
  Mass: Density=1
}
