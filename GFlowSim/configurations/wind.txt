// Granular "wind tunnel" simulation
// For small radii, e.g. 0.0125-0.015, needed a max dt of 0.0002.

// This is a 2d simulation
Dimensions: 2

// Set the simulation bounds
Bounds: {
  : -Front, Length  // X
  : -Height, Height // Y
}

// VARIABLES
Var: MinR=0.04
Var: MaxR=0.05
Var: Velocity=2.2
Var: Length=18
Var: Height=6
Var: Front=6

Integrator: VelocityVerlet

Boundary: {
  : Wrap
  : Repulse
}

// Set the number of particle types
NTypes: 1

// Creates a random force network of HardSphere and Lennard-Jones forces
Force-grid: Hertz

// Fill an area with particles
Fill: Area {
  // Fill all the simulation with particles
  Bounds: Full

  Template: TypeA {
    Radius: Uniform {
      Min: MinR
      Max: MaxR
    }
    Mass: Density=1
    Type: 0
  }

  Number: Phi=0.925 {
    TypeA : 1
  }

  Velocity: Zero
}

Particle {
  Position: 0, 0
  Sigma: 1
  Velocity: 0, 0
  Mass: inf
}

Reconcile: Remove=1.

Modifier: WindTunnel=Velocity
