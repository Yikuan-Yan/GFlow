// This is a 2d simulation
Dimensions: 2

// Set the simulation bounds
Bounds: {
  : 0, 4 // X
  : 0, 4 // Y
}

Integrator: VelocityVerlet

// Set the number of particle types
NTypes: 2

// Creates a random force network of HardSphere and Lennard-Jones forces
Force-grid: Random

// Fill an area with particles
Fill-area {
  // Fill all the simulation with particles
  Bounds: Full

  Template: TypeA {
    Radius: 0.05
    Mass: Density=1
    Type: 0
  }

  Template: TypeB {
    Radius: Uniform {
      Min: 0.02
      Max: 0.04
    }
    Mass: Density=1
    Type: 1
  }

  Number: Phi=0.5 {
    TypeA : 1
    TypeB : 1
  }

  Relax: 0.25
}
