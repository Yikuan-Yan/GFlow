// Variables
Var: dimensions=2
Var: width=32
Var: fraction=0.125
Var: phi=0.5
Var: h=0.5
Var: rP=0.05
Var: rC=0.01

Value: length = fraction*width

Dimensions: dimensions

// Set the simulation bounds
Bounds: Box=width

Integrator: VelocityVerlet {
  MaxDT: 0.0001
}

Boundary: Wrap

// Set the number of particle types
NTypes: 2

// Creates a random force network of HardSphere and Lennard-Jones forces
Force-grid: {
  : 0, 0, HardSphere
  : 0, 1, HardSphere
  : 1, 1, None
}

Fill: Polymer {
  Parallel: 1
  H: h  
  Length: length
  R: rP
  r: rC
  Phi: phi
}

// Fill an area with particles
Fill: Area {
  // Fill all the simulation with particles
  Bounds: Full
  
  Template: TypeA {
    Radius: 0.05
    Mass: Density=1
    Type: 1
  }

  Number: Phi=0.01 {
    TypeA : 1
  }
  
}
